CREATE TABLE AUTORES (

    Pais VARCHAR,
    url_autor varchar, 
    bibliografia varchar not null
);

CREATE TABLE MANGAS (
    Cod_manga SERIAL NOT NULL PRIMARY KEY,
    Nome VARCHAR,
    Sinopse VARCHAR,
    Preco_unit DECIMAL NOT NULL,
    Quantidade NUMERIC NOT NULL,
    Paginas NUMERIC NOT NULL,
    Url_capa VARCHAR NOT NULL,
    banner SERIAL NOT NULL,
    Genero SERIAL NOT NULL,
    Autor SERIAL NOT NULL
);
